\chapter{Discussion}

\section{Limitations}

\begin{itemize}
    \item Complexity of Social Media Images - Diverse Content, Contexts and Formats
    \item Technical Constraints - Size of modern Transformer Models (Image Captioning Models) is mostly too large
    \item Subjectivity of existing Dataset Ground Truths
    \item No Social Media Specific Datasets for Salient Object Detection
\end{itemize}

\section{Future Research}

\begin{itemize}
    \item Development of Social Media Specific Salient Object Detection Datasets
    \item Exploration of Lightweight Models for Social Media Image Analysis
    \item Integration of Multimodal Data (e.g., Text and Image) for Enhanced Salient Object Detection (use image captions as additional input)
    \item Look into further applications of Salient Object Detection in Social Media Analysis
\end{itemize}
